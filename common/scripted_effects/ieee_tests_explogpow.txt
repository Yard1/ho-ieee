ieee_test_assert_equal_or_nan = {
    set_temp_variable = { test_failed = 0 }
    if = {
        limit = {
            NOT = { check_variable = { actual_output = expected_output } }
            OR = {
                set_temp_variable = { arg_a = actual_output }
                NOT = { ieee_isnan = yes }
                set_temp_variable = { arg_a = expected_output }
                NOT = { ieee_isnan = yes }
            }
        }
        set_temp_variable = { test_failed = 1 }
    }
}
ieee_test_assert_close = {
    set_temp_variable = { test_failed = 0 }
    set_temp_variable = { arg_a = actual_output }
    if = {
        limit = { ieee_isnan = yes }
        set_temp_variable = { test_failed = 1 }
    }

    set_temp_variable = { arg_a = actual_output }
    set_temp_variable = { arg_b = expected_output }
    ieeeSub = yes
    set_temp_variable = { diff = out_a }

    if = {
        limit = {
            set_temp_variable = { arg_a = diff }
            set_temp_variable = { arg_b = 0 }
            compareLess = yes
        }
        set_temp_variable = { arg_a = diff }
        set_temp_variable = { arg_b = -1082130.432 }
        ieeeMul = yes
        set_temp_variable = { diff = out_a }
    }

    if = {
        limit = {
            set_temp_variable = { arg_a = diff }
            set_temp_variable = { arg_b = tolerance }
            compareGreater = yes
        }
        set_temp_variable = { test_failed = 1 }
    }
}
ieee_test_log_result = {
    if = {
        limit = { check_variable = { test_failed = 1 } }
        log = "Test [?test_id] - FAIL"
        add_to_temp_array = { test_fail_array = test_id }
    }
    else = {
        log = "Test [?test_id] - PASS"
    }

    set_temp_variable = { to_print_float = actual_output }
    log = "Actual: [print_float_full]"
    set_temp_variable = { to_print_float = expected_output }
    log = "Expect: [print_float_full]"
}
